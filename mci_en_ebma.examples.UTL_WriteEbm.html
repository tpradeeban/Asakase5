Example of using UTL_WriteEbm()

This is an example of loading the EBM up with 8 bit data where the data starts at 0x0, increment to 0xF, and then repeat this process. 



    unsigned int dataInput[ ] = {
       0x0, 0x1, 0x2, 0x3, 0x4, 0x5, 0x6, 0x7, 0x8, 0x9, 0xa, 0xb, 0xc, 0xd, 0xe, 0xf};

    unsigned int ebmData[4];
    unsigned int ebmDataCount = 4;

    ebmData[0] = 0x3 << 24 | 0x2 << 16 | 0x1 << 8 | 0x0;  /* -> 0x03020100 */
    ebmData[0] = 0x7 << 24 | 0x6 << 16 | 0x5 << 8 | 0x4;  /* -> 0x07060504 */
    ebmData[0] = 0xb << 24 | 0xa << 16 | 0x9 << 8 | 0x8;  /* -> 0x0b0a0100 */
    ebmData[0] = 0xf << 24 | 0xe << 16 | 0xd << 8 | 0xc;  /* -> 0x0f0e0d0c */

    EbmAccessHandle ebmAH;

    ebmAH = UTL_GetEbmAccessHandle();
    UTL_SetEbmAccessStartIndex(ebmAH, 0);
    UTL_WriteEbm    (ebmAH, ebmData, sizeof(unsigned int) * ebmDataCount);
    UTL_DeleteHandle(ebmAH);
