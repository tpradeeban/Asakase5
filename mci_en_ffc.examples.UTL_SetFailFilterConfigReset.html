Example of using UTL_SetFailFilterConfigReset()



    FailFilterConfigHandle ffCH;

    ffCH = UTL_GetFailFilterConfigHandle();

    UTL_SetFailFilterConfigAction                   (ffCH, UT_FF_UBM);
    UTL_SetFailFilterConfigDutBitSize               (ffCH, BIT_SIZE);
    UTL_SetFailFilterConfigBitSize                  (ffCH, BIT_SIZE);
    UTL_SetFailFilterConfigPinList                  (ffCH, 0, "TARGET_DQs", UT_ON, 1);
    UTL_SetFailFilterConfigNibbleMode               (ffCH, UT_FF_NIBBLE_DISABLE);
    UTL_SetFailFilterConfigBlockSize                (ffCH, PAGE_COUNT);
    UTL_SetFailFilterConfigStatusBytes              (ffCH, 0);
    UTL_SetFailFilterConfigPageSize                 (ffCH, COLUMN_COUNT);

    if (LT > 0)
        UTL_SetFailFilterConfigLowThreshold         (ffCH, countMethod, LT);
    UTL_SetFailFilterConfigHighThreshold            (ffCH, countMethod, HT);
    UTL_SetFailFilterConfigRepairRegionHighThreshold(ffCH, countMethod, RR);
    UTL_SetFailFilterConfigTFCHighThreshold         (ffCH, countMethod, TFC);
    UTL_SetFailFilterConfigMode                     (ffCH, UT_FF_LT, UT_ON);
    UTL_SetFailFilterConfigFailListCounterSelect    (ffCH, UT_FF_ALL_CNTRS, UT_ON);
    UTL_SetFailFilterConfigTFCOverlayCount          (ffCH, UT_OFF);
    UTL_SetFailFilterConfigECCSectors               (ffCH, ECC_SECTORS_PER_PAGE,
        (COLUMN_COUNT)/ECC_SECTORS_PER_PAGE, 0);
    UTL_SetFailFilterConfigRepairRegion             (ffCH, ECC_SECTORS_PER_PAGE/2,
        (COLUMN_COUNT)/(ECC_SECTORS_PER_PAGE/2), 0);

    /* Here, execute address assignments */

    UTL_SetFailFilterConfigReset                    (ffCH, UT_ON);
    UTL_SendFailFilterConfig                        (ffCH);
