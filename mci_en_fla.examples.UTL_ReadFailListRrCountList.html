Example of using UTL_ReadFailListRrCountList()

A page can be subdivided into a number of repair regions where a repair region is made up of some multiple of ECC Sectors. Refer to UTL_SetFailFilterConfigECCSectors().

Below is an example to read each repair region counter value over some range of pages.

If there are 4 pages, 2 planes, and 2 repair regions per plane, there would be 4 x 2 x 2 = 16 repair region counter values in total. The values would be accessed like this: 



    pagex[ 0] = 0,  pagey[ 0] = 0,  datalist[ 0] = Repair Region counter value 0 for page 0, plane 0
    pagex[ 1] = 0,  pagey[ 1] = 0,  datalist[ 1] = Repair Region counter value 1 for page 0, plane 0
    pagex[ 2] = 0,  pagey[ 2] = 0,  datalist[ 2] = Repair Region counter value 0 for page 0, plane 1
    pagex[ 3] = 0,  pagey[ 3] = 0,  datalist[ 3] = Repair Region counter value 1 for page 0, plane 1
    pagex[ 4] = 1,  pagey[ 4] = 0,  datalist[ 4] = Repair Region counter value 0 for page 1, plane 0
    pagex[ 5] = 1,  pagey[ 5] = 0,  datalist[ 5] = Repair Region counter value 1 for page 1, plane 0
    pagex[ 6] = 1,  pagey[ 6] = 0,  datalist[ 6] = Repair Region counter value 0 for page 1, plane 1
    pagex[ 7] = 1,  pagey[ 7] = 0,  datalist[ 7] = Repair Region counter value 1 for page 1, plane 1
    pagex[ 8] = 2,  pagey[ 8] = 0,  datalist[ 8] = Repair Region counter value 0 for page 2, plane 0
    pagex[ 9] = 2,  pagey[ 9] = 0,  datalist[ 9] = Repair Region counter value 1 for page 2, plane 0
    pagex[10] = 2,  pagey[10] = 0,  datalist[10] = Repair Region counter value 0 for page 2, plane 1
    pagex[11] = 2,  pagey[11] = 0,  datalist[11] = Repair Region counter value 1 for page 2, plane 1
    pagex[12] = 3,  pagey[12] = 0,  datalist[12] = Repair Region counter value 0 for page 3, plane 0
    pagex[13] = 3,  pagey[13] = 0,  datalist[13] = Repair Region counter value 1 for page 3, plane 0
    pagex[14] = 3,  pagey[14] = 0,  datalist[14] = Repair Region counter value 0 for page 3, plane 1
    pagex[15] = 3,  pagey[15] = 0,  datalist[15] = Repair Region counter value 1 for page 3, plane 1




    USlider pagecount = 0, valuecount = 0;
    USlider *pagex, *pagey, *datalist, *overflowlist;
    /*
      pagecount  is the total number of pages within the range being checked.

      valuecount is the total number of values to be returned.
      If 256 pages, 2 planes, 2 repair regions per plane, valuecount is 1024.
     */

    FailListAccessHandle flAH:
    DutCursor dutcsr;
    UT_DUT dut;

    flAH = UTL_GetFailListAccessHandle();

    UTL_SetFailListAccessTestSelect   (flAH, UT_FL_TESTSEL_NAME);
    UTL_ClearFailListAccessTestName   (flAH);
    UTL_AddFailListAccessTestName     (flAH, testname);
    UTL_SetFailListAccessDeviceChannel(flAH, 1);
    UTL_SetFailListAccessAllBit       (flAH, UT_ON);

    dutcsr = UTL_GetDutCursor(UT_DDUT);
    while ( (dut = UTL_NextDut(dutcsr)) != UT_NOMORE ) {
        UTL_SetFailListAccessDut  (flAH, dut);

        UTL_SetFailListAccessXAddr(flAH, 0x0, PAGES_PER_BLOCK * BLOCK_COUNT - 1);
        UTL_SetFailListAccessYAddr(flAH, 0,   COLUMNS_PER_PAGE - 1);

        UTL_SetFailListAccessRecordType(flAH, UT_FL_RECORD_COUNT_RR);

        /* Obtain the values for pagecount and valuecount */
        UTL_ReadFailListRecordCount(flAH, &pagecount, &valuecount);

        pagex = NULL, pagey = NULL, datalist = NULL, overflowlist = NULL;

        if (pagecount > 0) {
            pagex = malloc(pagecount * sizeof(USlider));           /* Array to hold page numbers   */
            pagey = malloc(pagecount * sizeof(USlider));           /* Array to hold column numbers */
        }
        if (valuecount > 0) {
            datalist     = malloc(valuecount * sizeof(USlider));   /* Array to hold count values   */
            overflowlist = malloc(valuecount * sizeof(USlider));   /* Array to hold overflowlist   */
        }

        /* Read pagex and pagey           using pagecount  to specify how many to read */
        /* Read datalist and overflowlist using valuecount to specify how many to read */
        UTL_ReadFailListRrCountList(flAH, pagex, pagey, pagecount, datalist, overflowlist, valuecount);

        int page, plane, num_counters_per_page;

        /* For multi-plane, there will be multiple sets of RR counts per page */
        num_counters_per_page = valuecount / pagecount;

        for (page = 0; page < pagecount; page++) {
            printf("page[%d] x=%d y=%d (data, overflow)\n", page, pagex[page], pagey[page]);

            for (plane = 0; plane < num_counters_per_page; plane++) {
                printf("(%d %d) ", datalist    [page * num_counters_per_page + plane],
                                   overflowlist[page * num_counters_per_page + plane]);
            }
            printf("\n");
        }

        if (pagex)        { free(pagex);        pagex        = NULL; }
        if (pagey)        { free(pagey);        pagey        = NULL; }
        if (datalist)     { free(datalist);     datalist     = NULL; }
        if (overflowlist) { free(overflowlist); overflowlist = NULL; }

    } /* dutcsr*/

    UTL_DeleteCursor(dutcsr);
    UTL_DeleteHandle(flAH);
