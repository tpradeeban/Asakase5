Example of using UTL_ReadFailListBadBlockCount()



    while ( (dut = UTL_NextDut(dutcsr)) != UT_NOMORE ) {

        UTL_SetFailListAccessDut(flAH, dut);

        /********************************
            Bad Block Determination
         ********************************

         Store the number of bad blocks that were detected in countBadBlock[dut-1].

         The value of argument 'status' will be either:
         0: UT_FL_COUNT_STATUS_COMPLETE
         1: UT_FL_COUNT_STATUS_OVERFLOW

         */

        countBadBlock[dut-1] = UTL_ReadFailListBadBlockCount(flAH, &status);
    }
